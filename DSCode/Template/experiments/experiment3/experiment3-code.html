#include <stdio.h>
#define MAX 100 // Define the maximum size of the queue

int queue[MAX];
int front = -1, rear = -1;

// Function to insert an element into the queue (enqueue)
void enqueue(int item) {
    if (rear == MAX - 1) {
        printf("Overflow: Queue is full\n");
        return;
    }
    if (front == -1) {
        front = 0;
    }
    rear++;
    queue[rear] = item;
    printf("%d inserted into the queue\n", item);
}

// Function to remove an element from the queue (dequeue)
int dequeue() {
    if (front == -1 || front > rear) {
        printf("Underflow: Queue is empty\n");
        return -1;
    }
    int val = queue[front];
    if (front == rear) {
        front = rear = -1; // Reset queue if last element is removed
    } else {
        front++;
    }
    printf("%d removed from the queue\n", val);
    return val;
}

// Function to return the front element of the queue
int top() {
    if (front == -1 || front > rear) {
        printf("Underflow: Queue is empty\n");
        return -1;
    }
    return queue[front];
}

// Function to display the elements in the queue
void display() {
    if (rear == -1) {
        printf("Empty queue\n");
        return;
    }
    printf("Queue elements: ");
    for (int i = front; i <= rear; i++) {
        printf("%d ", queue[i]);
    }
    printf("\n");
}

// Main function to demonstrate queue operations
int main() {
    int choice, item;
    do {
        printf("\nQueue Operations:\n");
        printf("1. Enqueue\n2. Dequeue\n3. Top\n4. Display\n5. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                printf("Enter the element to insert: ");
                scanf("%d", &item);
                enqueue(item);
                break;
            case 2:
                dequeue();
                break;
            case 3:
                item = top();
                if (item != -1) {
                    printf("Front element: %d\n", item);
                }
                break;
            case 4:
                display();
                break;
            case 5:
                printf("Exiting...\n");
                break;
            default:
                printf("Invalid choice! Please try again.\n");
        }
    } while (choice != 5);

    return 0;
}
